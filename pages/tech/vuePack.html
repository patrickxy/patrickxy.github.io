//title/index.js
// 导入组件，组件必须声明 name
import patrickTitle from "./main";

// 为组件提供 install 安装方法，供按需引入
patrickTitle.install = function(Vue) {
  Vue.component(patrickTitle.name, patrickTitle);
};
// 默认导出组件
export default patrickTitle;

//packages/index.js
// 导入单个组件
import patrickTitle from "./title/index";
import patrickTodoList from "./todoList/index";

// 以数组的结构保存组件，便于遍历
const components = [patrickTitle, patrickTodoList];

// 定义 install 方法
const install = function(Vue) {
  if (install.installed) return;
  install.installed = true;
  // 遍历并注册全局组件
  components.map(component => {
    Vue.component(component.name, component);
  });
};

if (typeof window !== "undefined" && window.Vue) {
  install(window.Vue);
}

export default {
  // 导出的对象必须具备一个 install 方法
  install,
  // 组件列表
  ...components
};

# 修改 vue.config.js

module.exports = {
  pages: {
    index: {
      entry: "src/main.js",
      template: "public/index.html",
      filename: "index.html"
    }
  },
  // 组件样式内联
  css: {
    extract: false
  },
  // 扩展 webpack 配置，使 packages 加入编译
  chainWebpack: config => {
    config.module
      .rule("js")
      .include.add("/packages/")
      .end()
      .use("babel")
      .loader("babel-loader")
      .tap(options => {
        // 修改它的选项...
        return options;
      });
  }
};

# 修改 packge.json 配置

{
"name": "patrick-componnets",
  "private": false,
  "version": "0.1.2",
  "license": "MIT",
  "scripts": {
    "serve": "vue-cli-service serve",
    "build": "vue-cli-service build",
    "lint": "vue-cli-service lint",
    "lib": "vue-cli-service build --target lib --name patrick-coms --dest lib packages/index.js""
  }，
  "main": "lib/patrick-coms.umd.min.js",
}

配置说明

  1. name npm 的包名，发布成功后可以在 npm 官网 上搜索

  2. private->false 设置为公开

  3. version 版本号 每次发布版本号必须更新

  4. 添加 license MIT

  5. scripts 中添加编译项 "lib": "vue-cli-service build --target lib --name patrick-coms --dest lib packages/index.js"

    --targer lib 代表打包成库文件

    --name patrick-coms 打包生成文件的名称

    patrick/index.js 代码的入口文件地址

  6. 添加 main: "lib/patrick-coms.umd.min.js" 指向打包后的入口文件

npm 包发布流程

  1. npm run lib
  2. 查看当前的 npm 源 npm config get registry npmjs.com
  3. npm login 然后输入账号密码 登录
  4. npm publish 发布，成功后命令行会有提示信息
  5. npm unpublish 撤回 npm 包

# 使用方法

使用说明

  1. npm install packagename

  2. import patrickComs from "patrick-components";

  3. Vue.use(patrickComs);

  4. 开始使用组件

    组件使用名称要跟 vue 文件中的 name 一致
